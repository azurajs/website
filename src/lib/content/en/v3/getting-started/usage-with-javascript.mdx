# Usage with JavaScript

While AzuraJS is designed as a TypeScript-first framework, it works perfectly fine with **plain JavaScript**. The functional routing API (`app.get()`, `app.post()`, etc.) provides an Express-like experience that requires no decorators, no type annotations, and no build step (when using Node.js with ESM or Bun).

<Callout type="info" title="Decorators require TypeScript">
  The decorator-based controller API (`@Controller`, `@Get`, `@Post`, etc.) requires TypeScript with `experimentalDecorators` enabled. If you want to use JavaScript only, use the **functional routing API** described on this page.
</Callout>

## Setup

### Prerequisites

- **Node.js 18+** or **Bun 1.0+**
- No TypeScript, no build tools required

### Installation

<Tabs items={["npm", "yarn", "pnpm", "bun"]}>
<Tab value="npm">
```bash
npm install azurats
```
</Tab>
<Tab value="yarn">
```bash
yarn add azurats
```
</Tab>
<Tab value="pnpm">
```bash
pnpm add azurats
```
</Tab>
<Tab value="bun">

```bash
bun add azurats
```
</Tab>

### Package.json

Set `"type": "module"` in your `package.json` to use ESM imports:

```json
{
  "name": "my-azura-app",
  "version": "1.0.0",
  "type": "module",
  "scripts": {
    "start": "node src/main.js",
    "dev": "node --watch src/main.js"
  },
  "dependencies": {
    "azurats": "^3.0.0"
  }
}
```

<Callout type="info" title="Node.js --watch mode">
  Node.js 18.11+ includes a built-in `--watch` flag that automatically restarts the process when files change. No additional dev dependencies needed.
</Callout>

## Basic Example

Create `src/main.js`:

```javascript
import { AzuraServer, CORSPlugin, HelmetPlugin, RateLimitPlugin, HttpError } from "azurats";

const app = new AzuraServer({
  environment: "development",
  server: { port: 3000 },
  logging: { level: "debug", requestLogging: true },
  debug: true,
});

// ── Plugins ──────────────────────────────────
app.plugin(CORSPlugin({ origins: "*" }));
app.plugin(HelmetPlugin());
app.plugin(RateLimitPlugin({ windowMs: 60_000, max: 100 }));

// ── Routes ───────────────────────────────────
app.get("/", (req, res) => {
  res.json({
    message: "Hello from AzuraJS v3!",
    timestamp: new Date().toISOString(),
  });
});

// ── Start ────────────────────────────────────
app.listen(3000, () => {
  console.log("Server running on http://localhost:3000");
});
```

Run it:

```bash
node src/main.js
```

That's it — no compilation, no build step.

## CRUD API Example

Here's a complete CRUD example using the functional routing API:

```javascript
import {
  AzuraServer,
  CORSPlugin,
  HelmetPlugin,
  RateLimitPlugin,
  RequestIdPlugin,
  CompressionPlugin,
  HealthCheckPlugin,
  HttpError,
} from "azurats";

const app = new AzuraServer({
  environment: "development",
  server: { port: 3000 },
  logging: { level: "debug", requestLogging: true },
});

// ── Plugins ──────────────────────────────────
app.plugin(RequestIdPlugin());
app.plugin(HelmetPlugin());
app.plugin(CORSPlugin({ origins: "*", credentials: false }));
app.plugin(RateLimitPlugin({ windowMs: 60_000, max: 100 }));
app.plugin(CompressionPlugin({ threshold: 1024 }));
app.plugin(
  HealthCheckPlugin({
    path: "/health",
    checks: {
      memory: () => process.memoryUsage().heapUsed < 500 * 1024 * 1024,
    },
  }),
);

// ── In-memory data store ─────────────────────
let nextId = 3;
const users = [
  { id: 1, name: "Alice", email: "alice@example.com" },
  { id: 2, name: "Bob", email: "bob@example.com" },
];

// ── Routes ───────────────────────────────────

// Root
app.get("/", (req, res) => {
  res.json({
    message: "Welcome to AzuraJS v3.0.0",
    timestamp: new Date().toISOString(),
  });
});

// List users with pagination
app.get("/users", (req, res) => {
  const page = Number(req.query.page) || 1;
  const limit = Number(req.query.limit) || 10;
  const start = (page - 1) * limit;

  res.json({
    users: users.slice(start, start + limit),
    pagination: { page, limit, total: users.length },
  });
});

// Get single user
app.get("/users/:id", (req, res) => {
  const user = users.find((u) => u.id === Number(req.params.id));
  if (!user) {
    throw HttpError.notFound("User not found");
  }
  res.json(user);
});

// Create user
app.post("/users", async (req, res) => {
  const { name, email } = req.body ?? {};
  if (!name || !email) {
    throw HttpError.badRequest("name and email are required");
  }

  const user = { id: nextId++, name, email };
  users.push(user);
  res.status(201).json(user);
});

// Update user
app.put("/users/:id", async (req, res) => {
  const user = users.find((u) => u.id === Number(req.params.id));
  if (!user) {
    throw HttpError.notFound("User not found");
  }

  const body = req.body ?? {};
  if (body.name) user.name = body.name;
  if (body.email) user.email = body.email;

  res.json(user);
});

// Delete user
app.delete("/users/:id", (req, res) => {
  const index = users.findIndex((u) => u.id === Number(req.params.id));
  if (index === -1) {
    throw HttpError.notFound("User not found");
  }
  users.splice(index, 1);
  res.status(204).send("");
});

// ── Error Handler ────────────────────────────
app.onError((err, req, res) => {
  if (err instanceof HttpError) {
    res.status(err.statusCode).json(err.toJSON());
    return;
  }
  console.error("Unhandled error:", err);
  res.status(500).json({
    error: { statusCode: 500, message: "Internal Server Error" },
  });
});

// ── Start ────────────────────────────────────
app.listen(3000, () => {
  console.log("Server running on http://localhost:3000");
});
```

## Routing API Reference

The functional routing API mirrors the Express pattern:

```javascript
app.get(path, ...handlers);
app.post(path, ...handlers);
app.put(path, ...handlers);
app.delete(path, ...handlers);
app.patch(path, ...handlers);
app.head(path, ...handlers);
app.options(path, ...handlers);
app.all(path, ...handlers);     // matches all HTTP methods
```

### Route Parameters

Use `:param` syntax for dynamic segments:

```javascript
app.get("/users/:id", (req, res) => {
  console.log(req.params.id); // "42" for /users/42
});

app.get("/posts/:postId/comments/:commentId", (req, res) => {
  console.log(req.params.postId);    // "10"
  console.log(req.params.commentId); // "5"
});
```

### Query Parameters

Access query string parameters through `req.query`:

```javascript
// GET /search?q=hello&page=2
app.get("/search", (req, res) => {
  console.log(req.query.q);    // "hello"
  console.log(req.query.page); // "2"
});
```

### Request Body

The body is automatically parsed for `POST`, `PUT`, `PATCH`, and `DELETE` requests:

```javascript
app.post("/data", async (req, res) => {
  console.log(req.body); // parsed JSON body
  res.json({ received: req.body });
});
```

### Request Properties

Every request handler receives `req` (AzuraRequest) and `res` (AzuraResponse):

```javascript
app.get("/debug", (req, res) => {
  res.json({
    method: req.method,          // "GET"
    path: req.pathname,          // "/debug"
    query: req.query,            // { key: "value" }
    params: req.params,          // {}
    headers: req.headers,        // { host: "localhost:3000", ... }
    cookies: req.cookies,        // { session: "abc123" }
    ip: req.ip,                  // "127.0.0.1"
    protocol: req.protocol,      // "http"
    secure: req.secure,          // false
    hostname: req.hostname,      // "localhost"
  });
});
```

### Response Methods

| Method | Description |
|---|---|
| `res.json(data)` | Send a JSON response with `Content-Type: application/json`. |
| `res.send(body)` | Send a plain text or Buffer response. |
| `res.html(body)` | Send an HTML response with `Content-Type: text/html`. |
| `res.status(code)` | Set the HTTP status code. Chainable. |
| `res.redirect(url, code?)` | Redirect to another URL (default 302). |
| `res.header(name, value)` | Set a response header. Chainable. |
| `res.type(contentType)` | Set the `Content-Type` header. Chainable. |
| `res.cookie(name, value, opts?)` | Set a cookie. Chainable. |
| `res.clearCookie(name, opts?)` | Clear a cookie. Chainable. |
| `res.attachment(filename?)` | Set `Content-Disposition` to attachment. Chainable. |

Example with chaining:

```javascript
app.get("/download", (req, res) => {
  res.status(200)
    .header("X-Custom", "value")
    .type("text/csv")
    .attachment("data.csv")
    .send("id,name\n1,Alice\n2,Bob");
});
```

## Middleware

In JavaScript, middleware works the same way — functions with `(req, res, next)` signatures:

```javascript
function logger(req, res, next) {
  console.log(`${req.method} ${req.pathname}`);
  next();
}

function requireApiKey(req, res, next) {
  const key = req.headers["x-api-key"];
  if (key !== "my-secret-key") {
    res.status(401).json({ error: "Invalid API key" });
    return;
  }
  next();
}

// Global middleware
app.use(logger);

// Route-specific middleware (pass before the handler)
app.get("/protected", requireApiKey, (req, res) => {
  res.json({ secret: "data" });
});
```

## Inline Middleware per Route

Pass multiple handlers to any route — all but the last are treated as middleware:

```javascript
const validate = (req, res, next) => {
  if (!req.body?.name) {
    throw HttpError.badRequest("name is required");
  }
  next();
};

const audit = (req, res, next) => {
  console.log(`Audit: ${req.method} ${req.pathname} by ${req.ip}`);
  next();
};

app.post("/users", validate, audit, async (req, res) => {
  res.status(201).json({ id: Date.now(), ...req.body });
});
```

## Error Handling

AzuraJS provides the `HttpError` class for structured HTTP errors:

```javascript
import { HttpError } from "azurats";

// Static factory methods
throw HttpError.badRequest("Invalid input");         // 400
throw HttpError.unauthorized("Login required");       // 401
throw HttpError.forbidden("Access denied");           // 403
throw HttpError.notFound("Resource not found");       // 404
throw HttpError.conflict("Resource already exists");  // 409

// Or create with a custom status code
throw new HttpError(422, "Validation failed");
```

Register a global error handler:

```javascript
app.onError((err, req, res) => {
  if (err instanceof HttpError) {
    res.status(err.statusCode).json(err.toJSON());
    return;
  }

  console.error(err);
  res.status(500).json({
    error: { statusCode: 500, message: "Internal Server Error" },
  });
});
```

The `err.toJSON()` method returns:

```json
{
  "error": {
    "statusCode": 404,
    "message": "Resource not found"
  }
}
```

## Using Plugins

All 15 built-in plugins work identically in JavaScript:

```javascript
import {
  AzuraServer,
  CORSPlugin,
  HelmetPlugin,
  RateLimitPlugin,
  JWTPlugin,
  SessionPlugin,
  CompressionPlugin,
  StaticPlugin,
  ETagPlugin,
  RequestIdPlugin,
  TimeoutPlugin,
  HealthCheckPlugin,
  CircuitBreakerPlugin,
  SSEPlugin,
  ProxyPlugin,
  MultipartPlugin,
} from "azurats";

const app = new AzuraServer();

// Security
app.plugin(HelmetPlugin());
app.plugin(CORSPlugin({ origins: ["https://mysite.com"] }));

// Performance
app.plugin(CompressionPlugin({ threshold: 512 }));
app.plugin(ETagPlugin());

// Rate limiting
app.plugin(RateLimitPlugin({ windowMs: 60_000, max: 100 }));

// Request tracking
app.plugin(RequestIdPlugin());
app.plugin(TimeoutPlugin({ timeout: 10_000 }));

// Static files
app.plugin(StaticPlugin({ root: "./public", prefix: "/static" }));

// Health check
app.plugin(HealthCheckPlugin({ path: "/health" }));
```

## JWT Authentication Example

```javascript
import { AzuraServer, JWTPlugin, signJWT, verifyJWT, HttpError } from "azurats";

const app = new AzuraServer();
const JWT_SECRET = "my-super-secret-key";

app.plugin(JWTPlugin({ secret: JWT_SECRET }));

// Login — issue a token
app.post("/login", async (req, res) => {
  const { username, password } = req.body ?? {};

  if (username !== "admin" || password !== "password") {
    throw HttpError.unauthorized("Invalid credentials");
  }

  const token = signJWT({ sub: username, role: "admin" }, JWT_SECRET, {
    expiresIn: "1h",
  });

  res.json({ token });
});

// Protected route
app.get("/profile", (req, res) => {
  const authHeader = req.headers["authorization"];
  if (!authHeader?.startsWith("Bearer ")) {
    throw HttpError.unauthorized("Missing token");
  }

  const token = authHeader.slice(7);
  const payload = verifyJWT(token, JWT_SECRET);

  if (!payload) {
    throw HttpError.unauthorized("Invalid token");
  }

  res.json({ user: payload.sub, role: payload.role });
});

app.listen(3000);
```

## JSDoc Type Hints

Even without TypeScript, you can get editor IntelliSense using JSDoc comments:

```javascript
/** @type {import('azurats/types').AzuraConfig} */
const config = {
  server: { port: 3000 },
  logging: { level: "debug" },
};

/**
 * @param {import('azurats/types').AzuraRequest} req
 * @param {import('azurats/types').AzuraResponse} res
 */
function handler(req, res) {
  res.json({ hello: "world" });
}

const app = new AzuraServer(config);
app.get("/", handler);
app.listen(3000);
```

This gives you autocomplete, parameter documentation, and type checking in editors like VS Code — no TypeScript compilation required.

## Config File in JavaScript

Create `azura.config.js` at the project root:

```javascript
/** @type {import('azurats/types').AzuraConfig} */
export default {
  environment: "development",
  server: {
    port: Number(process.env.PORT) || 3000,
    host: "0.0.0.0",
  },
  logging: {
    level: "debug",
    requestLogging: true,
  },
  debug: true,
};
```

## CommonJS Support

If your project uses CommonJS (`"type"` not set or set to `"commonjs"` in package.json), use `require`:

```javascript
const {
  AzuraServer,
  CORSPlugin,
  HelmetPlugin,
  HttpError,
} = require("azurats");

const app = new AzuraServer({
  server: { port: 3000 },
});

app.plugin(CORSPlugin({ origins: "*" }));
app.plugin(HelmetPlugin());

app.get("/", (req, res) => {
  res.json({ message: "Hello from CJS!" });
});

app.listen(3000);
```

<Callout type="warning" title="ESM recommended">
  While CommonJS is supported, we recommend **ESM** (`"type": "module"`) for new projects. ESM enables tree-shaking, top-level await, and is the standard module format for modern JavaScript.
</Callout>
